\draw[draw=black,fill=white] (3,3) circle circle (.3cm);

\draw[dashed,->] (0, 1.41412) arc(90:45:1.41412) node[above, yshift=3mm, xshift=-4mm]{$\theta$};
\draw[dashed,->] (0, -0.7) arc(270:135:0.7) node [left, xshift=-2mm, yshift=-2mm]{$\tau$};
